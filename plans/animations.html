// Loading animation with GSAP
function playLoadingAnimation() {
  // Reset animations
  gsap.killTweensOf("#loading-spinner .loading-circle");
  gsap.killTweensOf("#loading-spinner .loading-dot");
  
  // Set initial positions for dots
  gsap.set("#loading-spinner .loading-dot", { opacity: 0 });
  
  // Animate the circle
  gsap.to("#loading-spinner .loading-circle", {
    rotation: 360,
    duration: 1.5,
    repeat: 3,
    ease: "linear"
  });
  
  // Position and animate the dots
  const dots = document.querySelectorAll("#loading-spinner .loading-dot");
  dots.forEach((dot, index) => {
    const angle = (index / dots.length) * Math.PI * 2;
    const radius = 30;
    const x = Math.cos(angle) * radius;
    const y = Math.sin(angle) * radius;
    
    gsap.set(dot, { x, y });
    
    gsap.to(dot, {
      opacity: 1,
      delay: index * 0.15,
      duration: 0.3
    });
    
    gsap.to(dot, {
      opacity: 0,
      delay: 0.3 + index * 0.15 + 4,
      duration: 0.3
    });
  });
}

// Form validation animation with Mo.js
function playValidationAnimation() {
  const input = document.getElementById('validation-input');
  const value = input.value.trim();
  const resultElement = document.getElementById('validation-result');
  
  // Clear previous result
  resultElement.innerHTML = '';
  
  if (value.length > 0) {
    // Success animation
    const successIcon = document.createElement('i');
    successIcon.className = 'fas fa-check-circle';
    successIcon.style.color = 'var(--green)';
    successIcon.style.fontSize = '1.5rem';
    successIcon.style.marginLeft = '10px';
    successIcon.style.opacity = '0';
    resultElement.appendChild(successIcon);
    
    // Create burst animation
    const burst = new mojs.Burst({
      parent: successIcon,
      radius: { 0: 50 },
      count: 8,
      children: {
        shape: 'circle',
        fill: ['var(--green)', 'var(--teal)'],
        scale: { 1: 0 },
        radius: 10,
        duration: 500,
        easing: 'quad.out'
      }
    });
    
    // Show the icon with animation
    anime({
      targets: successIcon,
      opacity: 1,
      duration: 300,
      easing: 'easeOutCubic'
    });
    
    // Play burst animation
    burst.play();
    
    // Change input border
    anime({
      targets: input,
      borderColor: 'var(--green)',
      duration: 300
    });
  } else {
    // Error animation
    const errorIcon = document.createElement('i');
    errorIcon.className = 'fas fa-times-circle';
    errorIcon.style.color = '#ff5252';
    errorIcon.style.fontSize = '1.5rem';
    errorIcon.style.marginLeft = '10px';
    errorIcon.style.opacity = '0';
    resultElement.appendChild(errorIcon);
    
    // Show the icon with animation
    anime({
      targets: errorIcon,
      opacity: 1,
      duration: 300,
      easing: 'easeOutCubic'
    });
    
    // Shake animation for input
    anime({
      targets: input,
      translateX: [
        { value: -5, duration: 100 },
        { value: 5, duration: 100 },
        { value: -3, duration: 100 },
        { value: 3, duration: 100 },
        { value: 0, duration: 100 }
      ],
      borderColor: '#ff5252',
      duration: 500
    });
  }
}

// Simple game animation with GSAP
let gameRunning = false;
let gameScore = 0;
let jumpInProgress = false;
let obstacleAnimation;

function startGame() {
  if (gameRunning) return;
  
  // Reset game state
  gameRunning = true;
  gameScore = 0;
  document.getElementById('game-score').textContent = 'Score: 0';
  
  // Position elements
  gsap.set('#game-character', { y: 0 });
  gsap.set('#game-obstacle', { x: 0, right: -50 });
  
  // Start obstacle animation
  animateObstacle();
  
  // Add jump event listener
  document.getElementById('game-container').addEventListener('click', jumpCharacter);
  
  // Start score counter
  const scoreInterval = setInterval(() => {
    if (!gameRunning) {
      clearInterval(scoreInterval);
      return;
    }
    
    gameScore++;
    document.getElementById('game-score').textContent = `Score: ${gameScore}`;
  }, 100);
}

function animateObstacle() {
  // Get container width
  const containerWidth = document.getElementById('game-container').clientWidth;
  
  // Animate obstacle from right to left
  obstacleAnimation = gsap.to('#game-obstacle', {
    x: -containerWidth - 50,
    duration: 2 - Math.min(gameScore / 100, 1.5), // Speed increases with score
    ease: 'none',
    onComplete: function() {
      if (gameRunning) {
        // Reset position and run again
        gsap.set('#game-obstacle', { x: 0 });
        animateObstacle();
      }
    },
    onUpdate: checkCollision
  });
}

function jumpCharacter() {
  if (jumpInProgress || !gameRunning) return;
  
  jumpInProgress = true;
  
  // Jump animation
  gsap.to('#game-character', {
    y: -60,
    duration: 0.4,
    ease: 'power2.out',
    onComplete: function() {
      // Fall down
      gsap.to('#game-character', {
        y: 0,
        duration: 0.4,
        ease: 'power2.in',
        onComplete: function() {
          jumpInProgress = false;
        }
      });
    }
  });
}

function checkCollision() {
  const character = document.getElementById('game-character');
  const obstacle = document.getElementById('game-obstacle');
  
  // Get element positions
  const characterRect = character.getBoundingClientRect();
  const obstacleRect = obstacle.getBoundingClientRect();
  
  // Check for collision
  if (
    characterRect.right > obstacleRect.left &&
    characterRect.left < obstacleRect.right &&
    characterRect.bottom > obstacleRect.top &&
    characterRect.top < obstacleRect.bottom
  ) {
    endGame();
  }
}

function endGame() {
  gameRunning = false;
  
  // Stop obstacle animation
  if (obstacleAnimation) {
    obstacleAnimation.kill();
  }
  
  // Flash character red
  gsap.to('#game-character', {
    backgroundColor: '#ff5252',
    duration: 0.2,
    yoyo: true,
    repeat: 3
  });
}

document.getElementById('ripple-button').addEventListener('click', function(e) {
  // Create ripple element
  const ripple = document.createElement('span');
  ripple.classList.add('ripple');
  this.appendChild(ripple);
  
  // Set ripple position
  const rect = this.getBoundingClientRect();
  const size = Math.max(rect.width, rect.height);
  const x = e.clientX - rect.left;
  const y = e.clientY - rect.top;
  
  ripple.style.width = ripple.style.height = size + 'px';
  
  // Animate the ripple
  anime({
    targets: ripple,
    left: x + 'px',
    top: y + 'px',
    scale: [0, 1],
    opacity: [1, 0],
    duration: 800,
    easing: 'easeOutCubic',
    complete: function() {
      ripple.remove();
    }
  });
});

function playProgressAnimation() {
  // Reset progress
  gsap.set("#progress-indicator", { width: "0%" });
  gsap.set("#progress-percentage", { innerText: "0%" });
  
  // Animate to random percentage
  const targetProgress = Math.floor(Math.random() * 100) + 1;

    width: targetProgress + "%",
    duration: 1.5,
    ease: "power2.inOut"
  });

  let currentProgress = 0;
  const progressText = document.getElementById("progress-percentage");
  
  gsap.to(progressText, {
    duration: 1.5,
    ease: "power2.inOut",
    onUpdate: function() {
      currentProgress = Math.round(targetProgress * this.progress());
      progressText.innerText = currentProgress + "%";
    }
  });
}

let currentStep = 0;

function advanceStepProgress() {
  if (currentStep >= 4) {
    // Reset all steps
    currentStep = 0;
    gsap.set('#step-progress .step-circle', {
      backgroundColor: 'var(--medium-gray)'
    });
    gsap.set('#step-progress .step-line', {

    });
    return;

  currentStep++;

  const circles = document.querySelectorAll('#step-progress .step-circle');
  const lines = document.querySelectorAll('#step-progress .step-line');

    backgroundColor: 'var(--orange)',
    scale: [1, 1.2, 1],
    duration: 0.5,
    ease: 'back.out(1.7)'
  });

  if (currentStep > 1) {
    gsap.to(lines[currentStep - 2], {
      backgroundColor: 'var(--orange)',
      duration: 0.3,
      ease: 'power1.out'
    });
  }
}

// Star rating animation with Anime.js
function playStarRatingAnimation() {
  // Reset all stars
  const stars = document.querySelectorAll('#star-rating i');
  stars.forEach(star => {
    star.className = 'far fa-star';
    star.style.color = 'var(--medium-gray)';
  });
  
  // Generate random rating (1-5)
  const rating = Math.floor(Math.random() * 5) + 1;
  
  // Animate stars one by one
  for (let i = 0; i < rating; i++) {
    // Create animation timeline
    anime.timeline({ easing: 'easeOutElastic(1, .8)' })
      .add({
        targets: stars[i],
        scale: [1, 1.5, 1],
        duration: 600,
        delay: i * 150,
        update: function(anim) {
          // Change to solid star partway through animation

function initThreeScene() {
  // Create scene
  scene = new THREE.Scene();
  scene.background = new THREE.Color(0xf5f7fa);
  
  // Create camera
  camera = new THREE.PerspectiveCamera(
    75,
    document.querySelector('.three-container').clientWidth / 
    document.querySelector('.three-container').clientHeight,
    0.1,
    1000
  );
  camera.position.z = 5;
  
  // Create renderer
  renderer = new THREE.WebGLRenderer({ antialias: true });
  renderer.setSize(
    document.querySelector('.three-container').clientWidth,
    document.querySelector('.three-container').clientHeight
  );
  
  // Clear container and add renderer
  const container = document.getElementById('three-container');
  container.innerHTML = '';
  container.appendChild(renderer.domElement);
  
  // Create a colorful cube
  const geometry = new THREE.BoxGeometry(2, 2, 2);
  
  const materials = [
    new THREE.MeshBasicMaterial({ color: 0xffc067 }), // Orange
    new THREE.MeshBasicMaterial({ color: 0x66f4ff }), // Teal
    new THREE.MeshBasicMaterial({ color: 0xff8a5b }), // Coral
    new THREE.MeshBasicMaterial({ color: 0xffdc5b }), // Yellow
    new THREE.MeshBasicMaterial({ color: 0x64e291 }), // Green
    new THREE.MeshBasicMaterial({ color: 0x9b81e5 })  // Purple
  ];
  
  cube = new THREE.Mesh(geometry, materials);
  scene.add(cube);
  
  // Add subtle continuous rotation
  function animate() {
    requestAnimationFrame(animate);
    
    if (!animating) {
      cube.rotation.x += 0.005;
      cube.rotation.y += 0.005;
    }
    
    renderer.render(scene, camera);
  }
  
  animate();
  
  // Add mouse controls
  let isDragging = false;
  let previousMousePosition = {
    x: 0,
    y: 0
  };
  
  container.addEventListener('mousedown', function(e) {
    isDragging = true;
  });
  
  container.addEventListener('mousemove', function(e) {
    if (isDragging) {
      const deltaMove = {
        x: e.offsetX - previousMousePosition.x,
        y: e.offsetY - previousMousePosition.y
      };
      
      cube.rotation.y += deltaMove.x * 0.01;
      cube.rotation.x += deltaMove.y * 0.01;
    }
    
    previousMousePosition = {
      x: e.offsetX,
      y: e.offsetY
    };
  });
  
  container.addEventListener('mouseup', function(e) {
    isDragging = false;
  });
}

// Animate 3D object
function animateThreeObject() {
  if (animating) return;
  
  animating = true;
  
  // Store original rotation
  const originalRotation = {
    x: cube.rotation.x,
    y: cube.rotation.y,
    z: cube.rotation.z
  };
  
  // Perform a full spin animation
  gsap.to(cube.rotation, {
    x: cube.rotation.x + Math.PI * 2,
    y: cube.rotation.y + Math.PI * 2,
    duration: 1.5,
    ease: 'power2.inOut',
    onComplete: function() {
      animating = false;
    }
  });
}

// Create particles at position
function createParticles(x, y) {
  const playground = document.getElementById('particle-playground');
  const particleCount = 20 + Math.floor(Math.random() * 10);
  
  // Particle colors from brand palette
  const colors = [
    'var(--orange)',
    'var(--teal)',
    'var(--coral)',
    'var(--yellow)',
    'var(--green)',
    'var(--purple)'
  ];
  
  // Create particles
  for (let i = 0; i < particleCount; i++) {
    const particle = document.createElement('div');
    
    // Random particle properties
    const size = Math.random() * 10 + 5;
    const color = colors[Math.floor(Math.random() * colors.length)];

    particle.style.width = `${size}px`;
    particle.style.height = `${size}px`;
    particle.style.backgroundColor = color;
    particle.style.borderRadius = Math.random() > 0.5 ? '50%' : '3px';
